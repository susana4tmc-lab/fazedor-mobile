<script id="logicaAprende">

/* 
   ==========================================================
   VARIÁVEIS GLOBAIS PARA O MODO "APRENDE"
   ==========================================================
*/

let indiceAtual = 0;  
let progresso = {};   // guarda respostas já dadas

// Emoção "Alívio" só desbloqueia após resposta
function emocaoDisponivel(e) {
  if (!e.gated) return true;
  return progresso[indiceAtual]?.respostaSelecionada ? true : false;
}


/*
   ==========================================================
   RENDERIZAÇÃO DA FRASE ATUAL
   ==========================================================
*/

function renderFrase() {

  const container = document.getElementById("modo-aprende");
  container.innerHTML = ""; // limpar

  const dados = window.FRASES[indiceAtual];
  const total = window.FRASES.length;

  // CARD PRINCIPAL
  const card = document.createElement("div");
  card.className = "card";

  // Título
  const title = document.createElement("h3");
  title.textContent = dados.txt;

  // Categoria (opcional)
  const cat = document.createElement("div");
  cat.style.fontSize = "0.9rem";
  cat.style.color = "#666";
  cat.textContent = "Categoria: " + dados.categoria;

  // Barra de Emoções
  const emoBar = document.createElement("div");
  emoBar.className = "emoBar";

  window.EMOCOES.forEach(e => {
    const b = document.createElement("div");
    b.className = "emo";
    b.textContent = e.emoji;

    // Desativar "Alívio" até resposta existir
    if (!emocaoDisponivel(e)) {
      b.style.opacity = "0.4";
      b.style.pointerEvents = "none";
    }

    // Manter selecionadas
    if (progresso[indiceAtual]?.emoções?.includes(e.emoji)) {
      b.style.border = "2px solid var(--ok)";
    }

    b.onclick = () => {
      if (!progresso[indiceAtual]) progresso[indiceAtual] = {};
      if (!progresso[indiceAtual].emoções) progresso[indiceAtual].emoções = [];

      const arr = progresso[indiceAtual].emoções;

      // Toggle
      if (arr.includes(e.emoji)) {
        progresso[indiceAtual].emoções = arr.filter(x => x !== e.emoji);
        b.style.border = "1px solid var(--cinza-escuro)";
      } else {
        arr.push(e.emoji);
        b.style.border = "2px solid var(--ok)";
      }
    };

    emoBar.appendChild(b);
  });

  // Select de respostas
  const sel = document.createElement("select");

  const opt0 = document.createElement("option");
  opt0.value = "";
  opt0.textContent = "— escolher resposta —";
  sel.appendChild(opt0);

  dados.respostas.forEach(r => {
    const o = document.createElement("option");
    o.value = r;
    o.textContent = r;
    sel.appendChild(o);
  });

  // Se já havia resposta guardada
  if (progresso[indiceAtual]?.respostaSelecionada) {
    sel.value = progresso[indiceAtual].respostaSelecionada;
  }

  // Mensagem pedagógica (feedback)
  const feedback = document.createElement("div");
  feedback.style.marginTop = "12px";
  feedback.style.fontSize = "0.95rem";
  feedback.style.lineHeight = "1.3";

  function mostrarFeedback(texto, cor) {
    feedback.innerHTML = texto;
    feedback.style.color = cor;
    feedback.style.fontWeight = "600";
  }

  // Ao escolher resposta
  sel.onchange = () => {
    const val = sel.value;

    if (!progresso[indiceAtual]) progresso[indiceAtual] = {};
    progresso[indiceAtual].respostaSelecionada = val;

    // Desbloquear botão Próxima
    document.getElementById("btnNext").disabled = false;

    // Feedback pedagógico (Opção 3)
    mostrarFeedback(
      "✔️ Excelente escolha!<br>Esta resposta protege os teus limites, " +
      "mostra que sabes ouvir o que sentes e comunica com respeito e clareza.",
      "var(--ok)"
    );

    // Atualizar botão Alívio
    renderFrase(); // re-renderiza para permitir Alívio
  };

  // Navegação
  const nav = document.createElement("div");
  nav.className = "navBtns";

  const btnPrev = document.createElement("button");
  btnPrev.textContent = "⬅️ Anterior";
  btnPrev.onclick = () => {
    if (indiceAtual > 0) {
      indiceAtual--;
      renderFrase();
    }
  };
  if (indiceAtual === 0) btnPrev.disabled = true;

  const btnNext = document.createElement("button");
  btnNext.id = "btnNext";
  btnNext.textContent = "Próxima ➡️";
  btnNext.disabled = !progresso[indiceAtual]?.respostaSelecionada;
  btnNext.onclick = () => {
    if (indiceAtual < total - 1) {
      indiceAtual++;
      renderFrase();
    }
  };

  nav.appendChild(btnPrev);
  nav.appendChild(btnNext);

  // Indicador de progresso
  const prog = document.createElement("div");
  prog.style.fontSize = "0.85rem";
  prog.style.color = "#666";
  prog.style.marginTop = "10px";
  prog.textContent = `Frase ${indiceAtual + 1} de ${total}`;

  // Montar no card
  card.appendChild(title);
  card.appendChild(cat);
  card.appendChild(emoBar);
  card.appendChild(sel);
  card.appendChild(feedback);
  card.appendChild(nav);
  card.appendChild(prog);

  // Inserir no container
  container.appendChild(card);
}

/*
   ==========================================================
   FUNÇÃO PÚBLICA PARA MUDAR DE TAB (ligada ao BLOCO 1)
   ==========================================================
*/

function mudarTab(tab) {
  if (tab === "aprende") {
    document.getElementById("modo-aprende").style.display = "block";
    document.getElementById("modo-identifica").style.display = "none";

    document.querySelectorAll(".tab")[0].classList.add("active");
    document.querySelectorAll(".tab")[1].classList.remove("active");

  } else {
    document.getElementById("modo-aprende").style.display = "none";
    document.getElementById("modo-identifica").style.display = "block";

    document.querySelectorAll(".tab")[1].classList.add("active");
    document.querySelectorAll(".tab")[0].classList.remove("active");
  }
}

   ==========================================================
   INICIALIZAÇÃO
   ==========================================================
*/

document.addEventListener("DOMContentLoaded", () => {
  renderFrase();
});

</script>

const CATEGORIAS_IDENTIFICA = {
  "respeito": "Frases respeitadoras",
  "duvida": "Frases duvidosas",
  "pressao": "Frases de pressão"
};

// Estas são frases de "respeito" verdadeiras:
const RESPEITADORAS = [
  "Queres provar um bocadinho?",
  "Se não quiseres, não faz mal.",
  "Podes escolher.",
  "Está tudo bem se disseres que não.",
  "Queres ir brincar ou preferes ficar aqui?",
  "Diz se não te apetecer.",
  "Só se quiseres.",
  "Não faz mal nenhum se não quiseres.",
  "Faz como tu achares melhor",
  "Queres jogar comigo?"
];

// Estas são frases "duvidosas":
const DUVIDOSAS = [
  "Estou só a tentar ajudar…",
  "É para teu bem.",
  "Eu só quero ser simpático.",
  "Não percebo porque não queres."
];

// Todas as outras frases do MAP são de pressão
function tipoFrase(txt) {
  if (RESPEITADORAS.includes(txt)) return "respeito";
  if (DUVIDOSAS.includes(txt)) return "duvida";
  return "pressao";
}


/* ----------------------------------------------
   2. GERAR DECK PARA O EXERCÍCIO
   ---------------------------------------------- */

function gerarFrasesExercicio() {
  const press = window.FRASES.map(f => f.txt);

  const deck = [
    ...RESPEITADORAS.map(t => ({ txt: t, tipo: "respeito" })),
    ...DUVIDOSAS.map(t => ({ txt: t, tipo: "duvida" })),
    ...press.map(t => ({ txt: t, tipo: "pressao" }))
  ];

  // Embaralhar
  return deck.sort(() => Math.random() - 0.5);
}

/* ----------------------------------------------
   4. CRIAÇÃO DE DROPZONES
   ---------------------------------------------- */

function criarDropzone(tipo) {

  const container = document.createElement("div");
  container.className = "dropzone";

  const titulo = document.createElement("h4");
  titulo.textContent = CATEGORIAS_IDENTIFICA[tipo];
  container.appendChild(titulo);

  const holder = document.createElement("div");
  holder.className = "holder";
  holder.style.display = "flex";
  holder.style.flexDirection = "column";
  holder.style.gap = "8px";
  holder.style.marginTop = "10px";
  container.appendChild(holder);

  // Permitir largar
  container.ondragover = e => e.preventDefault();

  container.ondrop = e => {
    const txt = e.dataTransfer.getData("text");
    moverCardPara(tipo, txt, holder);
  };

  return { container, holder };
}


/* ----------------------------------------------
   5. CRIAR DRAGGABLE + BOTÕES ALTERNATIVOS
   ---------------------------------------------- */

function criarDraggable(obj, zonaRes, zonaDuv, zonaPre) {

  const d = document.createElement("div");
  d.className = "draggable";
  d.draggable = true;
  d.textContent = obj.txt;

  d.ondragstart = e => {
    e.dataTransfer.setData("text", obj.txt);
  };

  // Botões alternativos
  const btns = document.createElement("div");
  btns.style.display = "flex";
  btns.style.gap = "6px";
  btns.style.marginTop = "6px";

  const b1 = document.createElement("button");
  b1.textContent = "➡️ Respeito";
  b1.onclick = () => moverCardPara("respeito", obj.txt, zonaRes.holder);

  const b2 = document.createElement("button");
  b2.textContent = "➡️ Duvidosa";
  b2.onclick = () => moverCardPara("duvida", obj.txt, zonaDuv.holder);

  const b3 = document.createElement("button");
  b3.textContent = "➡️ Pressão";
  b3.onclick = () => moverCardPara("pressao", obj.txt, zonaPre.holder);

  [b1, b2, b3].forEach(b => {
    b.style.border = "none";
    b.style.padding = "6px";
    b.style.borderRadius = "8px";
    b.style.background = "var(--cinza)";
    b.style.cursor = "pointer";
  });

  btns.appendChild(b1);
  btns.appendChild(b2);
  btns.appendChild(b3);

  d.appendChild(btns);

  return d;
}


/* ----------------------------------------------
   6. MOVER CARTÃO PARA UMA CAIXA
   ---------------------------------------------- */

function moverCardPara(tipo, txt, destino) {

  // Remover qualquer cópia anterior
  document.querySelectorAll(".draggable").forEach(el => {
    if (el.childNodes[0].nodeValue === txt) el.remove();
  });

  // Criar novo card dentro da caixa
  const novo = document.createElement("div");
  novo.className = "draggable";
  novo.textContent = txt;

  destino.appendChild(novo);
}


/* ----------------------------------------------
   7. VERIFICAÇÃO DO EXERCÍCIO
   ---------------------------------------------- */

function verificarExercicio(zRes, zDuv, zPre) {

  let total = 0;
  let corretos = 0;

  // Função auxiliar
  function verificaZona(holder, tipoCerto) {
    const cards = holder.querySelectorAll(".draggable");
    cards.forEach(c => {
      total++;
      const texto = c.textContent.trim();
      const tipo = tipoFrase(texto);

      if (tipo === tipoCerto) {
        corretos++;
        c.style.border = "2px solid var(--ok)";
      } else {
        c.style.border = "2px solid var(--bad)";
      }
    });
  }

  verificaZona(zRes.holder, "respeito");
  verificaZona(zDuv.holder, "duvida");
  verificaZona(zPre.holder, "pressao");

  alert(`Acertaste ${corretos} em ${total}.`);
}


/* ----------------------------------------------
   8. MOSTRAR SOLUÇÃO
   ---------------------------------------------- */

function mostrarSolucao(zRes, zDuv, zPre, lista) {

  zRes.holder.innerHTML = "";
  zDuv.holder.innerHTML = "";
  zPre.holder.innerHTML = "";

  lista.forEach(obj => {
    const destino =
      obj.tipo === "respeito" ? zRes.holder :
      obj.tipo === "duvida"   ? zDuv.holder :
                                zPre.holder;

    const d = document.createElement("div");
    d.className = "draggable";
    d.textContent = obj.txt;
    destino.appendChild(d);
  });
}


/* ----------------------------------------------
   9. REFAZER SÓ AS ERRADAS
   ---------------------------------------------- */

function refazerErradas(zRes, zDuv, zPre, pool) {

  const recolher = [];

  function recolhe(holder, tipo) {
    const cards = holder.querySelectorAll(".draggable");
    cards.forEach(c => {
      const texto = c.textContent.trim();
      const t = tipoFrase(texto);

      if (t !== tipo) {
        recolher.push({ txt: texto, tipo: t });
        c.remove();
      }
    });
  }

  recolhe(zRes.holder, "respeito");
  recolhe(zDuv.holder, "duvida");
  recolhe(zPre.holder, "pressao");

  // Devolver as erradas ao pool
  recolher.forEach(obj => {
    pool.appendChild(criarDraggable(obj, zRes, zDuv, zPre));
  });
}

/* ==========================================================

function exportarTXT() {

  let linhas = [];

  for (let i = 0; i < window.FRASES.length; i++) {
    const f = window.FRASES[i];
    const prog = progresso[i];

    if (!prog || !prog.respostaSelecionada) continue;

    const emos = prog.emoções?.join(", ") || "(nenhuma)";

    linhas.push(
      `Frase: ${f.txt}\n` +
      `Categoria: ${f.categoria}\n` +
      `Emoções: ${emos}\n` +
      `Resposta escolhida: ${prog.respostaSelecionada}\n` +
      `----------------------------------------\n`
    );
  }

  if (linhas.length === 0) {
    alert("Ainda não respondeste a nenhuma frase.");
    return;
  }

  const blob = new Blob([linhas.join("")], { type: "text/plain;charset=utf-8" });
  const url = URL.createObjectURL(blob);

  const a = document.createElement("a");
  a.href = url;
  a.download = "respostas_assertivas.txt";
  a.click();

  URL.revokeObjectURL(url);
}
